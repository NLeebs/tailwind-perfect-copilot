  <action>
    Create `src/components/ShikiBlock.tsx` with EXACTLY this content. Do not add any other imports, props, or branches.

    ```tsx
    import { codeToHtml } from "shiki";

    const DARK_OVERRIDES = `.dark .shiki, .dark .shiki span {
      color: var(--shiki-dark) !important;
      background-color: var(--shiki-dark-bg) !important;
    }`;

    interface ShikiBlockProps {
      code: string;
      lang: "css" | "typescript" | "javascript";
    }

    export default async function ShikiBlock({
      code,
      lang,
    }: Readonly<ShikiBlockProps>) {
      const html = await codeToHtml(code, {
        lang,
        themes: {
          light: "github-light",
          dark: "github-dark",
        },
      });
      return (
        <>
          <style dangerouslySetInnerHTML={{ __html: DARK_OVERRIDES }} />
          <div
            className="overflow-x-auto rounded-xl text-sm leading-relaxed 3xl:text-base"
            dangerouslySetInnerHTML={{ __html: html }}
          />
        </>
      );
    }
    ```

    Implementation notes (do NOT deviate):
    - File MUST NOT contain `"use client"` — this is an async RSC.
    - The `!important` flag on the dark overrides is MANDATORY (RESEARCH.md Pitfall 1: Shiki's dual-theme output embeds inline `style=""` colors that beat class selectors without `!important`).
    - The selector is `.dark .shiki, .dark .shiki span` — plain CSS, NOT a Tailwind `dark:` prefix (the `<style>` tag content is raw CSS, not a Tailwind utility).
    - Themes must be exactly `github-light` (light) and `github-dark` (dark) per UI-SPEC §Color (Shiki themes).
    - Wrapper className must be exactly: `overflow-x-auto rounded-xl text-sm leading-relaxed 3xl:text-base`.
    - The `lang` type must be the union literal `"css" | "typescript" | "javascript"` — this is enforced for downstream call-site type safety.
    - Use `Readonly<ShikiBlockProps>` (mirroring CodeCallout's prop-destructure convention).

    After writing the file, run `yarn test --run` to confirm the new file does not break the existing test suite (the page that imports it is still the stub, so it should not yet be exercised — this is a smoke check).
